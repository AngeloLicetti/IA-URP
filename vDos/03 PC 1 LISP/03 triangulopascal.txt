(DEFUN PASCAL (X)
      (SETQ I 1 A '())
      (LOOP ((> I X) (RETURN "FIN DEL TRIANGULO"))
	    (SETQ J 0 S '())
	    (SETQ A (PASCAL-AUX A S I J))
            (PRINT A)
	    (SETQ I (+ I 1))
      )
)
(DEFUN PASCAL-AUX (A S I J)
       ((= J I) (RETURN S))
       ((OR (ZEROP J) (= J (- I 1))) (PUSH 1 S) (PASCAL-AUX A S I (+ J 1)))
       ((NOT (OR (ZEROP J) (= J (- I 1)))) (PUSH (+ (NTH J A) (NTH (- J 1) A)) S) (PASCAL-AUX A S I (+ J 1)))
)
